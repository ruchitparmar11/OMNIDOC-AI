086: def chunk_text(text, chunk_size=2000, overlap=300):
087:     chunks = []
088:     start = 0
089:     text_len = len(text)
090:     while start < text_len:
091:         end = min(start + chunk_size, text_len)
092:         chunks.append(text[start:end])
093:         start += (chunk_size - overlap)
094:     return chunks

137: def retrieve_relevant_chunks(question, history_ids, default_text, top_k=5):
138:     client_q = get_q_client()
139:     if not client_q:
140:         return default_text[:15000] # Fallback
141:         
142:     try:
143:         # We can pass one or multiple history_ids for Multi-Document Search
144:         if isinstance(history_ids, int):
145:             history_ids = [history_ids]
146:             
147:         model = get_embedder()
148:         question_embedding = model.encode([question], convert_to_numpy=True)[0]
149:         
150:         # 1. DENSE RETRIEVAL (Qdrant Database) - Fetches directly from disk!
151:         dense_top_k = 15
152:         search_result = client_q.search(
153:             collection_name="omnidoc_chunks",
154:             query_vector=question_embedding.tolist(),
155:             query_filter=Filter(
156:                 must=[
157:                     FieldCondition(
158:                         key="history_id",
159:                         match=MatchAny(any=history_ids)
160:                     )
161:                 ]
162:             ),
163:             limit=dense_top_k
164:         )
165:         
166:         if not search_result:
167:              return default_text[:15000]
168:              
169:         dense_chunks = [hit.payload['text'] for hit in search_result]
170:         
171:         # 2. SPARSE RETRIEVAL (BM25) - Done on the candidates
172:         from rank_bm25 import BM25Okapi
173:         import numpy as np
174:         tokenized_chunks = [chunk.lower().split() for chunk in dense_chunks]
175:         bm25 = BM25Okapi(tokenized_chunks)
176:         tokenized_query = question.lower().split()
177:         bm25_scores = bm25.get_scores(tokenized_query)
178:         sparse_indices = np.argsort(bm25_scores)[::-1].tolist()
179:         
180:         # 3. RECIPROCAL RANK FUSION (RRF)
181:         rrf_scores = {}
182:         rrf_k = 60
183:         
184:         for rank, hit in enumerate(search_result):
185:             # idx tracking within candidate list
186:             rrf_scores[rank] = rrf_scores.get(rank, 0) + 1 / (rrf_k + rank + 1)
187:             
188:         for rank, idx in enumerate(sparse_indices):
189:             rrf_scores[idx] = rrf_scores.get(idx, 0) + 1 / (rrf_k + rank + 1)
190:             
191:         hybrid_indices = sorted(rrf_scores.keys(), key=lambda x: rrf_scores[x], reverse=True)
192:         
193:         # 4. RERANKING (Cross-Encoder)
194:         rerank_model = get_reranker()
195:         cross_inp = [[question, dense_chunks[i]] for i in hybrid_indices]
196:         cross_scores = rerank_model.predict(cross_inp)
197:         
198:         reranked_pairs = sorted(zip(hybrid_indices, cross_scores), key=lambda x: x[1], reverse=True)
199:         final_top_indices = [pair[0] for pair in reranked_pairs[:top_k]]
200:         
201:         # Sort by chunk_index to keep chronlogical order from the document
202:         final_top_hits = [search_result[i] for i in final_top_indices]
203:         final_top_hits.sort(key=lambda hit: hit.payload.get('chunk_index', 0))
204:         
205:         relevant_context = "\n\n...[SNIP]...\n\n".join([hit.payload['text'] for hit in final_top_hits])
206:         return relevant_context
207:     except Exception as e:
208:         print(f"Advanced RAG Pipeline Error: {e}")
209:         return default_text[:15000]
